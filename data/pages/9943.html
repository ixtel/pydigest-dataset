simple relationship between 2 class (father & child)

class father(object):
    def __init__(self):
        super(father, self).__init__()
        self.father_name = "XR"
        self.father_age = 40

    def father_work(self):
        father_work = "IT"
        return father_work

class child(object):
    def __init__(self):
        super(child, self).__init__()
        self.child_name = "XG"
        self.child_age = 20

    def child_work(self):
        child_work = "STUDENTS"
        return child_work

class get_info(father, child):
    def __init__(self):
        super(get_info, self).__init__()
        print "What is ur name :", self.child_name ,"\n"
        print "What is ur father name :", self.father_name ,"\n"

        print "How old are u : I'm ", self.child_age ,"years old.\n"
