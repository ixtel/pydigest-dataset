\n from tkinter import *\n \n
\n import random\n \n
\n import time\n \n
\n \n\n \n
\n tk = Tk()\n \n
\n tk.title("Game")\n \n
\n tk.resizable(0, 0)\n \n
\n tk.wm_attributes("-topmost", 1)\n \n
\n canvas = Canvas(tk, width=100, height=400, bd=0, highlightthickness=0)\n \n
\n canvas.pack()\n \n
\n tk.update()\n \n
\n \n\n \n
\n class Ball:\n \n
\n def __init__(self, canvas, paddle, color):\n \n
\n self.canvas = canvas\n \n
\n self.paddle = paddle\n \n
\n self.id = canvas.create_oval(10, 10, 25, 25, fill=color)\n \n
\n self.canvas.move(self.id, 245, 100)\n \n
\n starts = [-3, -2, -1, 1, 2, 3]\n \n
\n random.shuffle(starts)\n \n
\n self.x = starts[0]\n \n
\n self.y = -3\n \n
\n self.canvas_height = self.canvas.winfo_height()\n \n
\n self.canvas_width = self.canvas.winfo_width()\n \n
\n self.hit_bottom = False\n \n
\n \n \n
\n def draw(self):\n \n
\n self.canvas.move(self.id, self.x, self.y)\n \n
\n pos = self.canvas.coords(self.id)\n \n
\n if pos[1] <= 0:\n \n
\n self.y = 3\n \n
\n if self.hit_paddle(pos) == True:\n \n
\n self.y = -3\n \n
\n if pos[3] >= self.canvas_height:\n \n
\n self.hit_bottom = True\n \n
\n if pos[0] <= 0:\n \n
\n self.x = 3\n \n
\n if pos[2] >= self.canvas_width:\n \n
\n self.x = -3\n \n
\n \n \n
\n def hit_paddle(self, pos):\n \n
\n paddle_pos = self.canvas.coords(self.paddle.id)\n \n
\n if pos[2] >= paddle_pos[0] and pos[0] <= paddle_pos[2]:\n \n
\n if pos[3] >= paddle_pos[1] and pos[3] <= paddle_pos[3]:\n \n
\n return True\n \n
\n return False\n \n
\n \n\n \n
\n class Paddle:\n \n
\n def __init__(self, canvas, color):\n \n
\n self.canvas = canvas\n \n
\n self.id = canvas.create_rectangle(0, 0, 100, 10, fill=color)\n \n
\n self.canvas.move(self.id, 200, 300)\n \n
\n self.x = 0\n \n
\n self.canvas_width = self.canvas.winfo_width()\n \n
\n self.canvas.bind_all('<KeyPress-Left>', self.turn_left)\n \n
\n self.canvas.bind_all('<KeyPress-Right>', self.turn_right)\n \n
\n \n \n
\n def turn_left(self, evt):\n \n
\n self.x = -2\n \n
\n \n \n
\n def turn_right(self, evt):\n \n
\n self.x = 2\n \n
\n \n \n
\n def draw(self):\n \n
\n self.canvas.move(self.id, self.x, 0)\n \n
\n pos = self.canvas.coords(self.id)\n \n
\n if pos[0] <= 0:\n \n
\n self.x = 0\n \n
\n elif pos[2] >= self.canvas_width:\n \n
\n self.x = 0\n \n
\n \n \n
\n paddle = Paddle(canvas, 'green')\n \n
\n ball = Ball(canvas, paddle, 'red')\n \n
\n \n\n \n
\n while 1:\n \n
\n if ball.hit_bottom == False:\n \n
\n ball.draw()\n \n
\n paddle.draw()\n \n
\n tk.update_idletasks()\n \n
\n tk.update()\n \n
