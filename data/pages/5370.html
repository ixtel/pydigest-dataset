#!/usr/bin/env python3
import sys
import requests
from lxml import html
import textwrap
def main(argv):
# Check usage.
if len(argv) == 1:
print("Error: expected a word or phrase for which to search.", file=sys.stderr)
return 1
# Consolidate the query phrase.
query = " ".join(argv[1::])
# Build the etymonline.com search url based on the URL API.
search_url_prefix = "http://etymonline.com/index.php?allowed_in_frame=0&search="
search_url_suffix = "&searchmode=term"
search_url = search_url_prefix + query + search_url_suffix
# Attempt to retrieve the HTML result page.
try:
request = requests.get(search_url)
except ConnectionError as error:
print("Error: failed to connect to etymonline.com: {}".format(error), file=sys.stderr)
return 1
# Parse the retrieved HTML and use an XPath query to find the "dictionary" division where results are presented.
parsed_html = html.fromstring(request.text)
dictionary_tree_list = parsed_html.body.xpath("//div[@id=\"dictionary\"]")
dictionary_tree = dictionary_tree_list[0][0] # It should be the only matching tag anyway.
# Handle the possibility of there being no results.
if len(dictionary_tree) == 0:
if dictionary_tree.text_content().strip() == "No matching terms found.":
print("Could not find any matching entries for \"{}\" in the etymonline.com database.".format(query))
return 0
else:
print("Error: unexpected result from etymonline.com.", file=sys.stderr)
return 1
# Extract the word entries and their etymologies with specific XPath queries.
entry_elements = dictionary_tree.xpath("//dt")
etymology_elements = dictionary_tree.xpath("//dd")
# Convert the internal HTML to raw text.
entries = [item.text_content() for item in entry_elements]
etymologies = [item.text_content() for item in etymology_elements]
# Word wrap potentially multi-paragraph etymologies. (Should probably have been built in...)
split_paragraph_etymologies = [etymology.split("\n") for etymology in etymologies]
wrap_and_rejoin = lambda split_paragraphs: "\n".join([textwrap.fill(paragraph) for paragraph in split_paragraphs])
etymologies = map(wrap_and_rejoin, split_paragraph_etymologies)
# Format the results to be displayed on a terminal.
formatted_results = ["\033[1m{}\033[0m\n\n{}".format(entry, etymology)
for entry, etymology in zip(entries, etymologies)]
print("\n\n".join(formatted_results))
return 0
if __name__ == "__main__":
sys.exit(main(sys.argv))
