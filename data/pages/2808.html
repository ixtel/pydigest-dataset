import os
from flask import Flask
from joe import joe

def handle_gitignores(names):
    '''Generates and sends the gitignore contents to stdout.'''
    exit = False
    output = '#### joe made this: http://goel.io/joe\n'
    for name in names:
        try:
            raw_name = joe.GITIGNORE_RAW[joe.GITIGNORE.index(name.lower())]
        except ValueError:
            print ('Uh oh! Seems like joe doesn\'t know what %s is.\n'
                   'Try running `joe ls` to see list of available gitignore '
                   'files.') % name
            exit = True
            break
        output += joe._fetch_gitignore(raw_name)
    return output

app = Flask(__name__)

@app.route('/<name>')
def get_gitignore(name):
  return handle_gitignores([name])

@app.route('/list')
def list():
  return ', '.join(joe.GITIGNORE)

@app.route('/')
def index():
  return joe.__doc__

if __name__ == '__main__':
	app.run( 
        host="0.0.0.0",
        port=int("80")
  )
