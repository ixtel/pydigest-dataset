import colorama

class Cell():
def __init__(self, piece=None):
self._piece = piece if piece else '?'
self._marked = False
def mark(self):
self._marked = True
def unmark(self):
self._marked = False
@property
def marked(self):
return self._marked
@property
def piece(self):
return self._piece
@piece.setter
def piece(self, value):
self._piece = value
def is_empty(self):
return self.piece == '_'
def is_full(self):
return not self.is_empty()
def __str__(self):
return self._piece.upper()
class MatchThreeGrid():
def _init_board(self):
self._board = [[Cell() for _ in range(self._width)] for _ in range(self._height)]
def __init__(self, text=None, h=None, w=None):
if text is not None:
self.parse_text(text)
elif h is not None and w is not None:
self._height = h
self._width = w
self._init_board()
else:
pass
self._last_move = (None, None, None)
def __getitem__(self, position):
assert type(position) == tuple
assert len(position) == 2
x, y = position
return self._board[y][x]
def __setitem__(self, position, value):
assert type(position) == tuple
assert len(position) == 2
assert type(value) == Cell
x, y = position
self._board[y][x] = value
def to_text(self):
result = ''
for y in range(self._height):
for x in range(self._width):
cell = self[x, y]
result += str(cell)
result += '\n'
return result
@property
def pieces_left(self):
pieces = 0
for y in range(self._height):
for x in range(self._width):
if self[x, y].is_full():
pieces += 1
return pieces
def display(self):
for y in range(self._height):
for x in range(self._width):
cell = self[x, y]
result = str(cell)
if (x, y) in self._last_move[0:2]:
result = colorama.Back.BLUE + result + colorama.Back.RESET
if cell.marked:
result = colorama.Fore.RED + result + colorama.Fore.RESET
print(result, end='')
print()
print()
def parse_text(self, text):
lines = text.splitlines()
self._height = len(lines)
self._width = len(lines[0])
self._init_board()
for y, line in enumerate(lines):
for x, piece in enumerate(line):
self[x, y].piece = piece
def swap(self, x, y, direction):
if direction == 'H':
x2, y2 = x + 1, y
elif direction == 'V':
x2, y2 = x, y + 1
else:
raise KeyError('Invalid direction: ' + direction)
if x2 >= self._width or y2 >= self._height:
return False
if self[x, y].piece == self[x2, y2].piece:
return False
self[x, y], self[x2, y2] = self[x2, y2], self[x, y]
self._last_move = ((x, y), (x2, y2), direction)
return True
def check_matches(self):
# unmark everyting
for y in range(self._height):
for x in range(self._width):
self[x, y].unmark()
found_match = False
# check horizontally
for y in range(self._height):
for x in range(self._height - 2):
pieces = [self[x + i, y].piece for i in range(3)]
if '_' in pieces:
continue
if pieces[0] == pieces[1] == pieces[2]:
found_match = True
for i in range(3):
self[x + i, y].mark()
# check vertically
for y in range(self._height - 2):
for x in range(self._height):
pieces = [self[x, y + i].piece for i in range(3)]
if '_' in pieces:
continue
if pieces[0] == pieces[1] == pieces[2]:
found_match = True
for i in range(3):
self[x, y + i].mark()
return found_match
def find_all_moves(self):
possible_moves = [[False for _ in range(self._width)] for _ in range(self._height)]
possible_moves_list = []
for y in range(self._height):
for x in range(self._width):
for direction in ('H', 'V'):
if self.swap(x, y, direction):
possible_moves[y][x] = self.check_matches()
possible_moves_list.append((x, y, direction))
self.swap(x, y, direction)
# c = msvcrt.getch()
self._last_move = (None, None, None)
return possible_moves_list
def find_legal_moves(self):
possible_moves = self.find_all_moves()
legal_moves = []
for move in possible_moves:
self.swap(*move)
if self.check_matches():
legal_moves.append(move)
self.swap(*move)
return legal_moves
def clear_won(self):
self._last_move = (None, None, None)
# destroy matched
for y in range(self._height):
for x in range(self._width):
if self[x, y].marked:
self[x, y].piece = '_'
self[x, y].unmark()
# shift rest down to cover gaps
for x in range(self._width):
while True:
shift_count = 0
for y in range(self._height - 1, 0, -1):
if self[x, y].is_empty() and self[x, y - 1].is_full():
self[x, y].piece, self[x, y - 1].piece = self[x, y - 1].piece, self[x, y].piece
shift_count += 1
if shift_count == 0:
break
class MatchThreeGame():
def __init__(self, initial_position):
self._initial_position = initial_position
self.movelist = []
self.is_final = True
self.board = MatchThreeGrid()
@property
def moves(self):
return ' '.join(['{}{}{}'.format(chr(ord('A') + move[0]), move[1], move[2].lower()) for move in self.movelist])
def display_moves(self):
print(self.moves, '\t', self.board.pieces_left)
def solve(self):
# self.display_moves()
self.board = MatchThreeGrid(text=self._initial_position)
legal_moves = self.board.find_legal_moves()
# print(legal_moves)
if len(legal_moves) == 0:
self.is_final = True
self.display_moves()
final_games.append(dict(win=self.board.pieces_left == 0,
moves=self.moves,
board=self.board.to_text(),
pieces_left=self.board.pieces_left))
if self.board.pieces_left == 0:
print('Found it!!!!')
return
for move in legal_moves:
# print(move)
self.board.swap(*move)
self.board.check_matches()
self.board.clear_won()
subgame = MatchThreeGame(self.board.to_text())
# board.display()
subgame.movelist = list(self.movelist)
subgame.movelist.append(move)
subgame.solve()
# reset board for next move
self.board.parse_text(self._initial_position)
