import math
import itertools
__author__ = "michaeljdavis"
"""
Find the most efficient way to pay off multiple loans
with different principals and rates
with a given available monthly payment.
"""
# Enter loans (principal, rate)
loans = {
"l1": (2013.65, .068),
"l2": (1585.68, .068),
"l3": (1989.62, .06),
"l4": (1604.41, .06),
"l5": (1824.11, .056),
"l6": (1163.39, .045),
"l7": (2331.34, .045)
}
# Enter amount you are willing to spend per month to payback loans
available_monthly_payment = 500.0
# Enter a range of percentages
# The greater the len(your_range), the longer this will take.
# Try to keep it as small as possible.
# The number of permutations grows quickly towards the hundreds of billions!
your_range = range(8, 23)
def main():
combinations = []
for a in itertools.product(your_range, repeat=len(loans)):
if sum(a) == 100:
print a
combinations.append(a)
print "-----------combo iterator complete------------"
print "-----------------please wait------------------"
print "-------calculating repayment efficiency-------"
combinations = [list(a) for a in combinations]
num_combinations = len(combinations)
def loan_payment_period(monthly_payment, principal, rate):
return ((math.log(monthly_payment)-math.log(monthly_payment-(principal*rate/12))) / math.log(1+rate/12))
results = {}
for combo in combinations:
combo_valid = True
while combo_valid:
# combo check
for n in range(7):
monthly_payment = available_monthly_payment * (combo[n]/100.0)
loan = loans['l{0}'.format(n+1)]
if monthly_payment - (loan[0]*loan[1]/12) < 0:
combo_valid = False
break
if not combo_valid:
break
max_payment_period = 0
for n in range(7):
monthly_payment = available_monthly_payment * (combo[n]/100.0)
loan = loans['l{0}'.format(n+1)]
payment_period = loan_payment_period(monthly_payment, loan[0], loan[1])
# print payment_period
if payment_period > max_payment_period:
max_payment_period = payment_period
result_key = max_payment_period
result_val = combo
print "{0} -> {1}".format(result_key, result_val)
results[result_key] = result_val
break
lowest_payment_period_combo = 9999.0
final_result = None
for key, val in results.iteritems():
if float(key) < lowest_payment_period_combo:
lowest_payment_period_combo = float(key)
print "{0} -> {1}".format(key, val)
final_result = [key, val]
print "----------------- \nBest repayment strategy:\n{0} -> {1}".format(round(final_result[0]), final_result[1])
for index, n in enumerate(final_result[1]):
monthly = available_monthly_payment*(n/100.0)
print "loan{0} ---> pay {1}/mo OR {2}/week OR {3}/day".format((index+1), monthly, round((monthly*12)/52), round((monthly*12)/365))
payment_period = loan_payment_period(monthly, loans["l{0}".format(index+1)][0], loans["l{0}".format(index+1)][1])
print "Repayement complete in {0} mos".format(round(payment_period))
print "Combinations tested ---> {0}".format(num_combinations)
if __name__ == '__main__':
main()
